#include <iostream>
#include <vector>
#include <string>
#include <sstream>
#include <functional>
#include <algorithm>
using namespace std;

class MiniVCS {
private:
    vector<string> ver;

    // Helper: split a stored version string into lines
    vector<string> splitLines(const string& s) const {
        vector<string> lines;
        stringstream ss(s);
        string line;
        while (getline(ss, line)) {
            lines.push_back(line);
        }
        return lines;
    }

public:
    // Saves a new version
    void save(const string& text) {
        ver.push_back(text);
        cout << "Version " << ver.size() << " saved.\n";
    }

    // Shows all versions with a short content preview
    void showVersions() const {
        if (ver.empty()) {
            cout << "No versions yet.\n";
            return;
        }
        cout << "\nVersions saved:\n";
        for (int i = 0; i < static_cast<int>(ver.size()); i++) {
            // Build a single-line preview (replace newlines with spaces)
            string preview = ver[i].substr(0, 60);
            replace(preview.begin(), preview.end(), '\n', ' ');
            if (ver[i].size() > 60) preview += "...";
            cout << "  Version " << (i + 1) << ": " << preview << "\n";
        }
    }

    // Returns the full text of a version; prints an error and returns ""
    // if the version number is out of range
    string getVersion(int num) const {
        if (num < 1 || num > static_cast<int>(ver.size())) {
            cerr << "Error: invalid version number " << num << ".\n";
            return "";
        }
        return ver[num - 1];
    }

    // Compares two versions line by line using LCS-based diff
    void diff(int a, int b) const {
        if (a < 1 || a > static_cast<int>(ver.size()) ||
            b < 1 || b > static_cast<int>(ver.size())) {
            cerr << "Error: one of the version numbers is out of range.\n";
            return;
        }

        const vector<string> A = splitLines(ver[a - 1]);
        const vector<string> B = splitLines(ver[b - 1]);
        const int m = static_cast<int>(A.size());
        const int n = static_cast<int>(B.size());

        // Build LCS (Longest Common Subsequence) DP table
        vector<vector<int>> dp(m + 1, vector<int>(n + 1, 0));
        for (int i = 1; i <= m; i++) {
            for (int j = 1; j <= n; j++) {
                if (A[i - 1] == B[j - 1]) {
                    dp[i][j] = dp[i - 1][j - 1] + 1;
                } else {
                    dp[i][j] = max(dp[i - 1][j], dp[i][j - 1]);
                }
            }
        }

        // Traceback to produce diff output
        function<void(int, int)> trace = [&](int i, int j) {
            if (i == 0 && j == 0) {
                return;
            } else if (i > 0 && j > 0 && A[i - 1] == B[j - 1]) {
                trace(i - 1, j - 1);
                cout << "  " << A[i - 1] << "\n";
            } else if (j > 0 && (i == 0 || dp[i][j - 1] >= dp[i - 1][j])) {
                trace(i, j - 1);
                cout << "+ " << B[j - 1] << "\n";
            } else {
                trace(i - 1, j);
                cout << "- " << A[i - 1] << "\n";
            }
        };

        cout << "\n--- Diff (Version " << a << " to Version " << b << ") ---\n";
        trace(m, n);
        cout << "--- End of Diff ---\n";
    }
};

int main() {
    MiniVCS vcs;

    vcs.save("Hello\nThis is version 1.");
    vcs.save("Hello\nThis is version 2.\nAdded line.");
    vcs.save("Hello there\nThis is version 3.\nAdded line.");

    vcs.showVersions();

    cout << "\n--- Restored Version 1 ---\n";
    string v1 = vcs.getVersion(1);
    if (!v1.empty()) cout << v1 << "\n";

    // Edge case: invalid version
    cout << "\n--- Attempting to restore Version 99 ---\n";
    vcs.getVersion(99);

    cout << "\n--- Difference between Version 1 and 2 ---\n";
    vcs.diff(1, 2);

    cout << "\n--- Difference between Version 2 and 3 ---\n";
    vcs.diff(2, 3);

    cout << "\n--- Difference between Version 1 and 3 ---\n";
    vcs.diff(1, 3);

    return 0;
}
